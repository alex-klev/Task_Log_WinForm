DROP table reference_bosses;

CREATE TABLE reference_bosses (
	boss_id         INTEGER PRIMARY KEY ASC ON CONFLICT ROLLBACK AUTOINCREMENT UNIQUE ON CONFLICT ROLLBACK NOT NULL ON CONFLICT ROLLBACK,
	data_id         TEXT NOT NULL ON CONFLICT ROLLBACK DEFAULT ('0'),
	fio             TEXT NOT NULL ON CONFLICT ROLLBACK,
	deleted_flg     INTEGER NOT NULL ON CONFLICT ROLLBACK DEFAULT (0),
	current_datetime DATETIME NOT NULL ON CONFLICT ROLLBACK DEFAULT (datetime('now', 'localtime'))
);

DROP table journal;

CREATE TABLE journal (
    journal_id       INTEGER PRIMARY KEY ASC 
                             ON CONFLICT ROLLBACK 
                             AUTOINCREMENT 
                             UNIQUE ON CONFLICT ROLLBACK 
                             NOT NULL ON CONFLICT ROLLBACK,
    data_id          TEXT NOT NULL ON CONFLICT ROLLBACK DEFAULT ('0'),
    boss_id          INTEGER NOT NULL ON CONFLICT ROLLBACK DEFAULT (1),
    employee_id      INTEGER NOT NULL ON CONFLICT ROLLBACK DEFAULT (1),
    task             TEXT NOT NULL ON CONFLICT ROLLBACK,
    note             TEXT,
    date_start_task  DATE NOT NULL ON CONFLICT ROLLBACK,
    date_end_task    DATE NOT NULL ON CONFLICT ROLLBACK,
    time_end_task    TIME,
    done             INTEGER NOT NULL ON CONFLICT ROLLBACK DEFAULT (0),
    deleted_flg      INTEGER NOT NULL ON CONFLICT ROLLBACK DEFAULT (0),
    current_datetime DATETIME NOT NULL ON CONFLICT ROLLBACK DEFAULT (datetime('now', 'localtime')),

    CONSTRAINT FK_journal_reference_bosses 
        FOREIGN KEY (boss_id) REFERENCES reference_bosses (boss_id) 
        ON DELETE SET DEFAULT     -- было CASCADE
        ON UPDATE CASCADE,

    CONSTRAINT FK_journal_reference_employes 
        FOREIGN KEY (employee_id) REFERENCES reference_employes (employee_id) 
        ON DELETE SET DEFAULT     -- было CASCADE
        ON UPDATE CASCADE
);

CREATE INDEX idx_journal_boss ON journal(boss_id);
CREATE INDEX idx_journal_employee ON journal(employee_id);
CREATE INDEX idx_date_end_task ON journal(date_end_task);
