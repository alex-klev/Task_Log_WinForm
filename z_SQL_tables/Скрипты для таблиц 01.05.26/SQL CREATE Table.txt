DROP table reference_bosses;

CREATE TABLE reference_bosses (
	boss_id         INTEGER PRIMARY KEY ASC ON CONFLICT ROLLBACK AUTOINCREMENT UNIQUE ON CONFLICT ROLLBACK NOT NULL ON CONFLICT ROLLBACK,
	data_id         TEXT NOT NULL ON CONFLICT ROLLBACK DEFAULT ('0'),
	fio             TEXT NOT NULL ON CONFLICT ROLLBACK,
	deleted_flg     INTEGER NOT NULL ON CONFLICT ROLLBACK DEFAULT (0),
	current_datetime DATETIME NOT NULL ON CONFLICT ROLLBACK DEFAULT (datetime('now', 'localtime'))
);

DROP table reference_employes;

CREATE TABLE reference_employes (
	employee_id     INTEGER PRIMARY KEY ASC ON CONFLICT ROLLBACK AUTOINCREMENT UNIQUE ON CONFLICT ROLLBACK NOT NULL ON CONFLICT ROLLBACK,
	data_id         TEXT NOT NULL ON CONFLICT ROLLBACK DEFAULT ('0'),
	fio             TEXT NOT NULL ON CONFLICT ROLLBACK,
	deleted_flg     INTEGER NOT NULL ON CONFLICT ROLLBACK DEFAULT (0),
	current_datetime DATETIME NOT NULL ON CONFLICT ROLLBACK DEFAULT (datetime('now', 'localtime'))
);

DROP table journal;

CREATE TABLE journal (
	journal_id       INTEGER PRIMARY KEY ASC ON CONFLICT ROLLBACK AUTOINCREMENT UNIQUE ON CONFLICT ROLLBACK NOT NULL ON CONFLICT ROLLBACK,
	data_id          TEXT NOT NULL ON CONFLICT ROLLBACK DEFAULT ('0'),
	boss_id          INTEGER REFERENCES reference_bosses (boss_id) ON DELETE CASCADE ON UPDATE CASCADE NOT NULL ON CONFLICT ROLLBACK,
	employee_id      INTEGER REFERENCES reference_employes (employee_id) ON DELETE CASCADE ON UPDATE CASCADE NOT NULL ON CONFLICT ROLLBACK,
	task             TEXT NOT NULL ON CONFLICT ROLLBACK,
	date_start_task  DATE NOT NULL ON CONFLICT ROLLBACK,
	date_end_task    DATE NOT NULL ON CONFLICT ROLLBACK,
	time_end_task    TIME NOT NULL ON CONFLICT ROLLBACK,
	done             INTEGER NOT NULL ON CONFLICT ROLLBACK DEFAULT (0),
	deleted_flg      INTEGER NOT NULL ON CONFLICT ROLLBACK DEFAULT (0),
	current_datetime DATETIME NOT NULL ON CONFLICT ROLLBACK DEFAULT (datetime('now', 'localtime')),
	CONSTRAINT FK_journal_reference_bosses FOREIGN KEY (boss_id) REFERENCES reference_bosses (boss_id) ON DELETE CASCADE ON UPDATE CASCADE,
	CONSTRAINT FK_journal_reference_employes FOREIGN KEY (employee_id) REFERENCES reference_employes (employee_id) ON DELETE CASCADE ON UPDATE CASCADE
);
