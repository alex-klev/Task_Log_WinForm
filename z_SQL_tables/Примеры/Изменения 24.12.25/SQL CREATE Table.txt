DROP TABLE debt_obligations;

CREATE TABLE debt_obligations (
	obligation_id      INTEGER PRIMARY KEY ASC ON CONFLICT ROLLBACK AUTOINCREMENT UNIQUE ON CONFLICT ROLLBACK NOT NULL ON CONFLICT ROLLBACK,
	data_id            TEXT NOT NULL ON CONFLICT ROLLBACK DEFAULT ('0'),
	user_id            INTEGER REFERENCES reference_users (user_id) ON DELETE CASCADE ON UPDATE CASCADE NOT NULL ON CONFLICT ROLLBACK,
	debt_name          TEXT NOT NULL ON CONFLICT ROLLBACK,
	bank               TEXT NOT NULL ON CONFLICT ROLLBACK,
	original_amount    NUMERIC(18,2) NOT NULL ON CONFLICT ROLLBACK DEFAULT (0.0),
	remaining_amount   NUMERIC(18,2) NOT NULL ON CONFLICT ROLLBACK DEFAULT (0.0),
	interest_rate      NUMERIC(18,2) NOT NULL ON CONFLICT ROLLBACK DEFAULT (0.0),
	payment_date       INTEGER NOT NULL ON CONFLICT ROLLBACK DEFAULT (0),
	period_payment     TEXT NOT NULL ON CONFLICT ROLLBACK,
	monthly_payment    NUMERIC(18,2) NOT NULL ON CONFLICT ROLLBACK DEFAULT (0.0),
	end_date           DATETIME NOT NULL ON CONFLICT ROLLBACK DEFAULT (datetime('now', 'localtime')),
	yearly_overpayment NUMERIC(18,2) NOT NULL ON CONFLICT ROLLBACK DEFAULT (0.0),
	deleted_flg        INTEGER NOT NULL ON CONFLICT ROLLBACK DEFAULT (0),
	current_datetime   DATETIME NOT NULL ON CONFLICT ROLLBACK DEFAULT (datetime('now', 'localtime')),
	CONSTRAINT FK_debt_obligations_reference_users FOREIGN KEY (user_id) REFERENCES reference_users (user_id) ON DELETE CASCADE ON UPDATE CASCADE
);

DROP TABLE assets;

CREATE TABLE assets (
	asset_id         INTEGER PRIMARY KEY ASC ON CONFLICT ROLLBACK AUTOINCREMENT UNIQUE ON CONFLICT ROLLBACK NOT NULL ON CONFLICT ROLLBACK,
	data_id          TEXT NOT NULL ON CONFLICT ROLLBACK DEFAULT ('0'),
	user_id          INTEGER REFERENCES reference_users (user_id) ON DELETE CASCADE ON UPDATE CASCADE NOT NULL ON CONFLICT ROLLBACK,
	bank             TEXT NOT NULL ON CONFLICT ROLLBACK,
	asset_type       TEXT NOT NULL ON CONFLICT ROLLBACK,
	currency         TEXT NOT NULL ON CONFLICT ROLLBACK,
	period           TEXT NOT NULL ON CONFLICT ROLLBACK,
	cash             NUMERIC(18,2) NOT NULL ON CONFLICT ROLLBACK DEFAULT (0.0),
	deleted_flg      INTEGER NOT NULL ON CONFLICT ROLLBACK DEFAULT (0),
	current_datetime DATETIME NOT NULL ON CONFLICT ROLLBACK DEFAULT (datetime('now', 'localtime')),
	CONSTRAINT FK_assets_reference_users FOREIGN KEY (user_id) REFERENCES reference_users (user_id) ON DELETE CASCADE ON UPDATE CASCADE
);
