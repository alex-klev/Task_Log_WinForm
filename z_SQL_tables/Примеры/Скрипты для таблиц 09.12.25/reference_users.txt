CREATE TABLE reference_users_new (
    user_id              INTEGER PRIMARY KEY ASC ON CONFLICT ROLLBACK AUTOINCREMENT UNIQUE ON CONFLICT ROLLBACK NOT NULL ON CONFLICT ROLLBACK,
    user_name            TEXT     NOT NULL ON CONFLICT ROLLBACK,
    email                TEXT     NOT NULL ON CONFLICT ROLLBACK UNIQUE ON CONFLICT ROLLBACK,
    phone_number         TEXT     NOT NULL ON CONFLICT ROLLBACK UNIQUE ON CONFLICT ROLLBACK,
    salt                 BLOB     NOT NULL ON CONFLICT ROLLBACK,
    hash                 BLOB     NOT NULL ON CONFLICT ROLLBACK,
    role_name            TEXT     NOT NULL ON CONFLICT ROLLBACK DEFAULT 'user',
    datetime_create_user DATETIME NOT NULL ON CONFLICT ROLLBACK DEFAULT (datetime('now', 'localtime'))
);
