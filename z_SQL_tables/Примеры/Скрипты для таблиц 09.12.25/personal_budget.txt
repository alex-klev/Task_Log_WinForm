CREATE TABLE personal_budget_new (
	pers_budget_id INTEGER PRIMARY KEY ASC ON CONFLICT ROLLBACK AUTOINCREMENT UNIQUE ON CONFLICT ROLLBACK NOT NULL ON CONFLICT ROLLBACK,
	data_id        TEXT NOT NULL ON CONFLICT ROLLBACK DEFAULT ('0'),
	user_id        INTEGER REFERENCES reference_users (user_id) ON DELETE CASCADE ON UPDATE CASCADE NOT NULL ON CONFLICT ROLLBACK,
	budget_name     TEXT NOT NULL ON CONFLICT ROLLBACK,
	cash_data       NUMERIC(18,2) NOT NULL ON CONFLICT ROLLBACK DEFAULT (0.0),
	budget_category TEXT NOT NULL ON CONFLICT ROLLBACK,
	budget_date    DATETIME NOT NULL ON CONFLICT ROLLBACK,
	deleted_flg    INTEGER NOT NULL ON CONFLICT ROLLBACK DEFAULT (0),
	current_datetime DATETIME NOT NULL ON CONFLICT ROLLBACK DEFAULT (datetime('now', 'localtime')),
	CONSTRAINT FK_personal_budget_reference_users FOREIGN KEY (user_id) REFERENCES reference_users (user_id) ON DELETE CASCADE ON UPDATE CASCADE
);
