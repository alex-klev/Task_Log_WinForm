CREATE VIEW additional_goals_view AS

SELECT ag.additional_goal_id, 
		ag.user_id, 
		ag.goal_name, 
		ag.goal_assessment, 
		ag.age, 
		ag.savings, 
		ag.years_left, 
		ag.capital, 
		ag.investing, 
		ag.investing_savings,
		ag.deleted_flg,
		ag.current_datetime
FROM additional_goals ag
WHERE ag.deleted_flg = 0;
