-- Пример "идеальной" настройки для веб-приложения

-- Включаем WAL режим для лучшей конкурентности
PRAGMA journal_mode = WAL;
-- Баланс производительности и безопасности
PRAGMA synchronous = NORMAL;
-- Ждем до 5 секунд, если БД занята
PRAGMA busy_timeout = 5000;
-- Увеличиваем кэш до 20 МБ
PRAGMA cache_size = -20000;
-- Включаем проверку внешних ключей
PRAGMA foreign_keys = ON;
-- Храним временные таблицы в памяти
PRAGMA temp_store = MEMORY;

------------------------------------------------
-— Увеличивает размер кэша (здесь до 80 МБ). 
-- Чем больше данных поместится в оперативной памяти, тем быстрее будут повторные запросы
PRAGMA cache_size = -80000;
